Overview

This report showcases the brand identity of the Center for Metropolitan Studies (CEM) and demonstrates the application of its brand.yml configuration.

Since the current CEM logo lacks consistency and color balance, the visual identity was designed based on CEM’s 20-Year Commemorative Logo.

The CEM brand files can be found in the brands folder in the .github repository.

Bar Plots

Code
penguins |>
  ggplot(aes(x = sex, fill = species)) +
  geom_bar(alpha = 0.8) +
  facet_wrap(~species, ncol = 1) +
  coord_flip() +
  labs(
    x = "Island",
    y = "Count"
  ) +
  scale_fill_brand_d()

Histograms

Code
penguins |>
  drop_na(flipper_length_mm, species) |>
  ggplot(aes(x = flipper_length_mm, fill = species)) +
  geom_histogram(bins = 30, alpha = 0.5, position = "identity") +
  labs(x = "Flipper Length (mm)", y = "Frequency", fill = "Species") +
  scale_fill_brand_d()

Boxplots

Code
penguins |>
  drop_na(bill_length_mm, species) |>
  ggplot(aes(x = species, y = bill_length_mm, fill = species)) +
  geom_boxplot(outlier.color = get_brand_color("dark-red")) +
  geom_jitter(width = 0.2, alpha = 0.1) +
  scale_fill_brand_d(alpha = 0.7) +
  labs(x = "Species", y = "Bill Length (mm)", fill = "Species")
#> ! The color dark-red was not found in the `_brand.yml` file.

Scatterplots

Code
penguins |>
  drop_na(body_mass_g, flipper_length_mm, species) |>
  ggplot(
    aes(
      x = body_mass_g,
      y = flipper_length_mm,
      color = species,
      shape = species
    )
  ) +
  geom_point(size = 2) +
  geom_smooth(method = "lm", formula = y ~ x, se = FALSE) +
  labs(
    x = "Body Mass (g)",
    y = "Flipper Length (mm)",
    color = "Species",
    shape = "Species"
  ) +
  scale_color_brand_d(alpha = 0.7)

Maps

Code
brazil_states_vector <- gadm("BRA", level = 1, path = data_dir)
Code
data <-
  tibble(
    state = sample(
      brazil_states_vector$NAME_1,
      size = 1000,
      replace = TRUE
    ),
    value = sample(1:1000, size = 1000, replace = TRUE)
  ) |>
   map_fill_data(col_fill = "value", col_ref = "state")

Continuous

Code
brazil_states_vector |>
  left_join(data, by = c("NAME_1" = "state")) |>
  ggplot() +
  geom_spatvector(aes(fill = value), color = get_brand_color("gray")) +
  scale_fill_brand_b(color_type = "seq") +
  labs(fill = NULL) +
  annotation_scale(
    location = "br",
    style = "tick",
    width_hint = 0.25,
    height = unit(0.15, "cm")
  ) +
  annotation_north_arrow(
    location = "br",
    height = unit(1, "cm"),
    width = unit(1, "cm"),
    pad_x = unit(0.1, "cm"),
    pad_y = unit(0.55, "cm"),
    style = north_arrow_fancy_orienteering
  )
